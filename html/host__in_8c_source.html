116int StoreReceiveFrame( uint8_t rx_byte ) {
117 const uint8_t STX = 0x02; // 開始バイト
118 const uint8_t ETX = 0x03; // 終了バイト
119 static uint8_t now_initial = 0; // 設定中の頭文字
120
121 inputUartState = UART_INPUT_ACTIVE; // 送信ステータス更新
122
123 /* STX受信でリセット */
124 if ( rx_byte == STX ) {
125 step = WAIT_FOR_STX;
126 }
127
128 /* 受信バイトをフォーマットと正しいかどうかを確認しながらフレームを作成 */
129 switch ( step ) {
130 /* --------- STXを受け取るとフレーム処理を開始 ---------------- */
131 case WAIT_FOR_STX:
132 if ( rx_byte == STX ) {
133 /* 処理を開始 */
134 InitHostInC();
135 step = CHECK_ETX_OR_NEW_INITIAL;
136 } else {
137 /* フレームエラー */
138 }
139 break;
140
141 /* -------- ETXまたは新しい頭文字を確認 ------------------------- */
142 case CHECK_ETX_OR_NEW_INITIAL:
143 if ( rx_byte == ETX ) {
144 /* ETX：更新処理 */
145 step = WAIT_FOR_STX;
146 inputUartState = UART_INPUT_COMPLETE; // 送信ステータス更新
147 return TRUE;
148 } else {
149 /* 新しい頭文字の処理 */
150 int result = isCharUniqueInFrame( rx_byte );
151 if ( result ) {
152 /* 定義あり重複なし*/
153 now_initial = rx_byte; // 頭文字の設定
154
155 if ( rx_byte == settings[CHANNEL].initial ) {
156 /* チャンネルの場合 */
157 step = PROCESS_CHANNEL;
158 } else {
159 step = PROCESS_OTHER;
160 }
161 } else {
162 /* 定義にない、あるいは重複*/
163 InitHostInC();
164 }
165 }
166 break;
167
168 /* -------- チャンネルの確認 ------------------------------------ */
169 case PROCESS_CHANNEL:
170 tmpChannelAscii[idxTmpChannel++] = rx_byte;
171
172 if ( idxTmpChannel == CHANNEL_LENGTH ) {
173 /* チャンネルを数字4桁に変換 */
174 uint16_t tmp_int_channel = convertIntegerToChannel( tmpChannelAscii );
175 /* チャンネル範囲の確認 */
176 int is_range_in_channel = isChannelInRange( tmp_int_channel );
177 if ( is_range_in_channel ) {
178 /* 範囲内 */
179 rxProperty[CHANNEL].tmpValue = tmp_int_channel;
180 rxProperty[CHANNEL].isUpdated = TRUE;
181 step = CHECK_ETX_OR_NEW_INITIAL;
182 } else {
183 InitHostInC();
184 }
185 }
186 break;
187
188 /* -------- その他設定値の確認 ------------------------------------- */
189 case PROCESS_OTHER:
190 if ( handleUpdateRxValue( now_initial, rx_byte ) ) {
191 /* 設定値の範囲内 */
192 step = CHECK_ETX_OR_NEW_INITIAL;
193 } else {
194 /* 設定値範囲外 */
195 InitHostInC();
196 }
197 break;
198
199 } // switch終了
200 return FALSE;
201}

212int UpdateData() {
213
214 int change_items = 0; // 更新があった項目数
215 int update_type = HANDSHAKE_NO_UPDATE;
216
217 /* 個々の設定値を更新 */
218 for ( int i = 0; i < PROPERTY_NUM; i++ ) {
219
220 if ( rxProperty[i].isUpdated == TRUE ) {
221 /* 入力があった設定の処理 */
222
223 if ( rxProperty[i].tmpValue == Get_currentValue( i ) ) {
224 /* 現在値と同じなら更新しない*/
225 continue;
226 }
227
228 if ( ( (Property)i == CHANNEL ) || ( (Property)i == ANTENNA ) ) {
229 /* チャンネルとアンテナ更新の場合 */
230 change_items++;
231 update_type = HANDSHAKE_REQUIRED;
232 rxProperty[i].isUpdated = TRUE;
233 } else {
234 /* それ以外 */
235 Set_currentValue( i, rxProperty[i].tmpValue );
236 if ( update_type != HANDSHAKE_REQUIRED ) {
237 update_type = HANDSHAKE_UPDATE_NO_REQ;
238 }
239 change_items++;
240 }
241 }
242 }
243
244 /* 設定値の変更項目数の集計 */
245 inputUartState = ( UART_INPUT_COMPLETE | change_items ); // 送信ステータス更新
246 return update_type;
247}

317/* チャンネル番号4桁をASCIIからINT型に変換して返す */
318int convertIntegerToChannel( uint8_t temp_char[] ) {
319
320 for ( int i = 0; i < CHANNEL_LENGTH; i++ ) {
321 if ( ( temp_char[i] < '0' ) || ( '9' < temp_char[i] ) ) {
322 /* 数値以外が入っている*/
323 return 0;
324 }
325 }
326
327 int num = temp_char[3] - '0'; // 1の位
328 num += ( temp_char[2] - '0' ) * 10; // 10の位
329 num += ( temp_char[1] - '0' ) * 100; // 100の位
330 num += ( temp_char[0] - '0' ) * 1000; // 1000の位
331
332 return num;
333}

336static int handleUpdateRxValue( uint8_t initial, uint8_t rx_value ) {
337
338 if ( ( rx_value < '0' ) || ( '9' < rx_value ) ) {
339 return 0;
340 }
341
342 /* ASCIIから数値に変換*/
343 uint8_t integer_val = rx_value - '0';
344
345 for ( int i = 0; i < PROPERTY_NUM; i++ ) {
346 if ( initial == settings[i].initial ) {
347 /* 頭文字の列を検索 */
348 if ( settings[i].rangeStart <= integer_val && integer_val <= settings[i].rangeEnd ) {
349 // 設定値が範囲内
350 rxProperty[i].tmpValue = integer_val;
351 rxProperty[i].isUpdated = TRUE;
352 return 1;
353 }
354 }
355 }
356 return 0; // 範囲外
357}
