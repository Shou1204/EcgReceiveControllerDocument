92/* フォーマット100バイトを作成 */
93static void generateTransFormat() {
94
95
96 OutputFormat[0] = START_BYTE;
97 OutputFormat[1] = DAT_NUM[0];
98 OutputFormat[2] = DAT_NUM[1];
99 OutputFormat[3] = FORMAT_VER;
100 OutputFormat[4] = HARD_VER;
101 OutputFormat[5] = SOFT_VER;
102 OutputFormat[6] = Get_paceMakerCnt();
103 OutputFormat[7] = Get_denAtuCnt();
104 OutputFormat[8] = Get_nurseCallCnt();
105 OutputFormat[9] = Get_readErrorCnt();
106
107 OutputFormat[10] = getTop7Bits( Get_currentValue( CHANNEL ) );
108 OutputFormat[11] = getBtm7Bits( Get_currentValue( CHANNEL ) );
109 OutputFormat[12] = (uint8_t)Get_currentValue( ANTENNA );
110 OutputFormat[13] = (uint8_t)Get_currentValue( POW_fFREQ );
111 OutputFormat[14] = (uint8_t)Get_currentValue( HUM_FILTER );
112 OutputFormat[15] = (uint8_t)Get_currentValue( DRIFT_FILTER );
113 OutputFormat[16] = (uint8_t)Get_currentValue( MUS_FILTER );
114 OutputFormat[17] = (uint8_t)Get_currentValue( MUS_ST );
115 OutputFormat[18] = (uint8_t)Get_currentValue( QRS_MODE );
116 OutputFormat[19] = (uint8_t)Get_currentValue( QRS_SENSI );
117 OutputFormat[20] = (uint8_t)Get_currentValue( ECG_SENSI );
118 OutputFormat[21] = (uint8_t)Get_currentValue( AUTO_QRS_ST );
119
120 OutputFormat[22] = Get_inputUartState();
121 OutputFormat[23] = handshakeState;
122 synchroState = (uint8_t)Get_syncSearchStatus();
123 OutputFormat[24] = synchroState;
124 OutputFormat[25] = Get_parityErr();
125 OutputFormat[26] = Get_chErr();
126
127 remainEcgBuf = GetRemainEcgBuf();
128 OutputFormat[27] = getTop7Bits( remainEcgBuf );
129 OutputFormat[28] = getBtm7Bits( remainEcgBuf );
130
131 nowTick = ( currentTick / 100 ) & 0x1FFFFFL;
132
133
134 OutputFormat[29] = ( nowTick & 0x1FC00 ) >> 14;
135 OutputFormat[30] = ( nowTick & 0x3F80 ) >> 7;
136 OutputFormat[31] = nowTick & 0x7F;
137 // output_7 = HAL_GetTick() / 100;
138 // OutputFormat[37] = output_7;
139
140 /* Ecgデータ */
141 for ( int i = 0; i < 30; i++ ) {
142 if ( IsReadEcg() == TRUE ) {
143 output_5++;
144 short tmp_ecg = GetEcgBuf( Get_currentValue( QRS_MODE ) );
145 // if ( tmp_ecg <= 1024 ) {
146 // printf( "tmp:%d\n", tmp_ecg );
147 // }
148 OutputFormat[( i * 2 ) + START_ECG_IDX] = getTop7Bits( (int)tmp_ecg );
149 OutputFormat[( i * 2 ) + 1 + START_ECG_IDX] = getBtm7Bits( (int)tmp_ecg );
150 } else {
151 /* Ecgバッファが足りない時 */
152 OutputFormat[( i * 2 ) + START_ECG_IDX] = getTop7Bits( 0x1100 );
153 OutputFormat[( i * 2 ) + 1 + START_ECG_IDX] = getBtm7Bits( 0x1100 );
154 }
155 }
156
157 OutputFormat[32] = output_1;
158 OutputFormat[33] = output_2;
159 OutputFormat[34] = output_3;
160 OutputFormat[35] = output_4;
161 OutputFormat[36] = output_5;
162 OutputFormat[37] = output_6;
163
164 OutputFormat[98] = checkSum( OutputFormat );
165 OutputFormat[99] = END_BYTE;
166}

168/* 加算チェックサムの下位7ビットを返す */
169static uint8_t checkSum( uint8_t *format ) {
170 int total = 0;
171 for ( int i = 0; i < 98; ++i ) {
172 total += format[i];
173 }
174 uint8_t result = total & 0x7F;
175 return result;
176}
