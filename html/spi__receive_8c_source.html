191int SpiReceiveProcess( uint8_t rx_byte ) {
192
193 /* どのバイト位置か追跡 */
194 bytePosition++;
195 if ( bytePosition >= FRAME_SIZE ) {
196 bytePosition = 0;
197 }
198
199 if ( syncGetCnt < SYNC_BUF_SIZE ) {
200 /* 同期用バッファ取り込み */
201 syncBuf[syncGetCnt++] = rx_byte;
202
203 if ( syncGetCnt == SYNC_BUF_SIZE ) {
204 /* バッファがいっぱい */
205 savedBytePosition = bytePosition;
206
207 if ( t2_syncro( syncBuf, &headBytes, &headBit ) ) {
208 /* 同期検出成功 */
209 syncSearchStatus = HEAD_SEARCH_SUCCESS;
210 // printf( "success!\n" );
211 } else {
212 syncSearchStatus = SYNC_FAIL;
213 syncGetCnt = 0;
214 printf( "fail\n" );
215 }
216 }
217 }
218
219 if ( syncSearchStatus == HEAD_SEARCH_SUCCESS ) {
220 /* 同期検出終了、歩調あわせ */
221 restartPosition = getRestartPosition( headBytes, savedBytePosition );
222 if ( restartPosition == bytePosition ) {
223 step5 = 4;
224 syncSearchStatus = ALREADY_SYNC;
225 continueErrsCnt = 0;
226 ecgOld = 1333;
227 printf( "///sync!!!\n" );
228 }
229 // T2PreNop = 1;
230 }
231
232
233 switch ( step5 ) {
234 /* -------------------------- step1 ---------------------------------*/
235 case 0:
236 /* 受信エラーフラグのリセット */
237 errorFlag = 0;
238
239 /* 前回のフレームを保存 */
240 savePreFrame( rx_byte );
241
242 /* 受信バイトを現フレームに格納 */
243 pic[step5] = rx_byte;
244 pic[5] = rx_byte;
245 step5 = 1;
246 break;
247 /* -------------------------- step1 ---------------------------------*/
248 case 1:
249 /* フィルタの設定 */
250 FilterSettings();
251
252 /* フィルタリングした値を作成 */
253 for ( int i = 0; i < 2; i++ ) {
254 long tmp_long_filtering = filteringEcgHandler( ecgLongData[i] );
255 filteringEcg[i] = PickEcgFromLong( tmp_long_filtering );
256 filteringStatus[i] = PickStsFromLong( tmp_long_filtering );
257 }
258
259 pic[step5] = rx_byte;
260 step5 = 2;
261 break;
262 /* -------------------------- step3 ---------------------------------*/
263 case 2:
264 /* QRS検出（自動）フィルタした値を作成 */
265 for ( int i = 0; i < 2; i++ ) {
266 autoEcg[i] = autoQrsHandler( filteringEcg[i], filteringStatus[i] );
267 }
268
269 pic[step5] = rx_byte;
270 step5 = 3;
271
272 break;
273 /* -------------------------- step3 ---------------------------------*/
274 case 3:
275 /* QRS検出（手動）フィルタした値を作成 */
276 for ( int i = 0; i < 2; i++ ) {
277 manualEcg[i] = manualQrsHandler( filteringEcg[i], filteringStatus[i] );
278 }
279
280 /* Ecgデータの格納 */
281 for ( int i = 0; i < 2; i++ ) {
282 StoreEcg( ecgLongData[i], autoEcg[i], manualEcg[i] );
283 }
284
285 pic[step5] = rx_byte;
286 step5 = 4;
287 break;
288 /* -------------------------- step4 ---------------------------------*/
289 case 4:
290 pic[step5] = rx_byte;
291 step5 = 0;
292
293 if ( syncSearchStatus != ALREADY_SYNC ) {
294 /*--- 完全な同期はずれ中 ---*/
295 for ( int i = 0; i < 2; i++ ) {
296 ecgLongData[i] = GenerateLongEcgData( 0x3FF, 0 );
297 }
298 // printf( "no_sync\n" );
299 output_2++;
300 break;
301 }
302
303 /* Ecgの抽出 */
304 ecg[0] = pickEcg1();
305 ecg[1] = pickEcg2();
306
307 /* 送信機ステータスの抽出 */
308 transmitterStatus = pickTranmitterState( prePic );
309
310 /* 送信機ステータスのカウント */
311 transmitterCnt( transmitterStatus );
312
313 /* フレームにエラーがあるか確認 */
314 errorFlag = frameErrorCheck();
315
316 if ( errorFlag != TRUE ) {
317 /*--- フレーム中に異常なし ---*/
318 continueErrsCnt = 0;
319 // ecgOld = ecg[1];
320 ecgOld = 5;
321 } else {
322 /*--- フレーム中にエラーがあった ---*/
323 output_1++;
324 for ( int i = 0; i < 2; i++ ) {
325 ecg[i] = ecgOld;
326 }
327
328 transmitterStatus = 0x02FF;
329
330 continueErrsCnt++;
331 if ( continueErrsCnt >= ERRORS_ ) {
332 /*--- 連続エラーのため同期検出処理へ ---*/
333 syncGetCnt = 0;
334 syncSearchStatus = SYNC_FAIL;
335 printf( "Error!!!\n" );
336 }
337 }
338
339 /* Ecgデータに送信機ステータスをヘッダーに加える */
340 for ( int i = 0; i < 2; i++ ) {
341 ecgLongData[i] = GenerateLongEcgData( transmitterStatus, ecg[i] );
342 }
343
344 break;
345 }
346#ifdef TESTCODE
347 /* テストコード用 */
348 debug_cnt++;
349#endif
350 return 0;
351}

358static int frameErrorCheck() {
359
360 int error_exists = FALSE;
361
362 /* チャンネルコードの比較 */
363 chCode = PickRxChannelCode();
364 if ( chCode != getChannelCode( Get_currentValue( CHANNEL ) ) ) {
365 /* チャンネル違い */
366 error_exists = TRUE;
367 chErr++;
368 }
369
370 /* パリティチェック */
371 if ( parityCheck() == FALSE ) {
372 error_exists = TRUE;
373 parityErr++;
374 }
375
376 return error_exists;
377}

441 /*
442 切替時に遅延の内容に調整

453 /* 自動QRS検出のデータ */

608/* チャンネルコードを取得 */
609static short PickRxChannelCode() {
610
611 long dat = 0x0000000L;
612 for ( int i = 0; i < 3; i++ ) {
613 dat <<= 8;
614 dat |= ( prePic[i + 1] & maskSet[headBit].Ch[i] );
615 }
616 dat >>= maskSet[headBit].ChShf;
617
618 short sync_byte = prePic[0] & maskSet[headBit].HeadBit; // 先頭のシンクロビットを取得
619 if ( sync_byte == 0 ) {
620 dat ^= 0x01FF;
621 } // '0' の時は反転
622
623 return (short)dat;
624}
