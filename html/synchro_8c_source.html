81int t2_syncro( uint8_t *ptr, int *m_bytes, int *m_bits ) {
82 unsigned char xorbuf[6], andbuf[6], orbuf[6];
83 unsigned char mask, bitpos;
84 unsigned char xortemp, andtemp1, ortemp1, andtemp2, ortemp2;
85 unsigned char wkptr1, wkptr2, wkptr3;
86 short base_cnt, cnt1, cnt2, cnt3;
87
88 base_cnt = 0;
89
90 for ( int i = 0; i < 5; i++ ) {
91 /*--- 5バイト分（1フレーム）のサーチ ---*/
92 xorbuf[i] = andbuf[i] = 0xFF;
93 orbuf[i] = 0;
94
95 // バッファ内の位置を設定
96 cnt1 = base_cnt + 0;
97 cnt2 = base_cnt + 5;
98 if ( cnt2 >= SYNC_BUF_SIZE ) cnt2 -= SYNC_BUF_SIZE;
99
100 cnt3 = base_cnt + 10;
101 if ( cnt3 >= SYNC_BUF_SIZE ) cnt3 -= SYNC_BUF_SIZE;
102
103 if ( ++base_cnt >= SYNC_BUF_SIZE ) base_cnt = 0;
104
105 for ( int j = 0; j < TEST_CNT; j++ ) {
106 /* （TEST_CNT × 2）フレーム分の検索を行う */
107 wkptr1 = ptr[cnt1];
108 wkptr2 = ptr[cnt2];
109 wkptr3 = ptr[cnt3];
110
111 /* 点滅しているビット列を抽出（3つのデータ間の変化を検出） */
112 xorbuf[i] &= ( ( wkptr1 ^ wkptr2 ) & ( wkptr2 ^ wkptr3 ) );
113
114 /* 少なくとも'1'が含まれるビット列を抽出（いずれかが1のビット） */
115 andbuf[i] &= ( ( wkptr1 & wkptr2 ) );
116
117 /* 少なくとも'1'が含まれたビット列を抽出 */
118 orbuf[i] |= ( ( wkptr1 | wkptr2 ) );
119
120 /* リングバッファの処理 */
121 cnt1 += 10;
122 if ( cnt1 >= SYNC_BUF_SIZE ) cnt1 -= SYNC_BUF_SIZE;
123 cnt2 += 10;
124 if ( cnt2 >= SYNC_BUF_SIZE ) cnt2 -= SYNC_BUF_SIZE;
125 cnt3 += 10;
126 if ( cnt3 >= SYNC_BUF_SIZE ) cnt3 -= SYNC_BUF_SIZE;
127 }
128 }
129
130 andbuf[5] = andbuf[0];
131 orbuf[5] = orbuf[0];
132
133 for ( int i = 0; i < 5; i++ ) {
134 xortemp = xorbuf[i];
135
136 // 点滅しているビットが存在する場合
137 if ( xortemp ) {
138 mask = 0x80;
139 andtemp1 = andbuf[i];
140 andtemp2 = andbuf[i + 1];
141 ortemp1 = orbuf[i];
142 ortemp2 = orbuf[i + 1];
143
144 // 点滅しているbit列の処理
145 for ( bitpos = 0; xortemp && mask; ( mask >>= 1, bitpos++ ) ) {
146 // 1があるbit列かどうか
147 if ( ( xortemp & mask ) ) {
148 // 上位5ビットの処理
149 if ( mask >= 0x08 ) {
150 // 特定のパターン（例えば'001'）に一致するか確認
151 if ( ( ( andtemp1 & MMU[bitpos] ) == MCU[bitpos] ) &&
152 ( ( ortemp1 & MMU[bitpos] ) == MCU[bitpos] ) ) {
153 // 同期パターンが見つかった場合、位置を設定してTRUEを返す
154 *m_bytes = i;
155 *m_bits = bitpos;
156 return TRUE;
157 }
158 } else {
159 // 下位3ビットの処理（次のバイトにまたがる場合）
160 if ( ( andtemp1 & MMU[bitpos] ) == MCU[bitpos] && ( ortemp1 & MMU[bitpos] ) == MCU[bitpos] &&
161 ( andtemp2 & MML[bitpos] ) == MCL[bitpos] && ( ortemp2 & MML[bitpos] ) == MCL[bitpos] )
162 {
163 // 同期パターンが見つかった場合、位置を設定してTRUEを返す
164 *m_bytes = i;
165 *m_bits = bitpos;
166 return TRUE;
167 }
168 }
169 // 処理済みのビットをクリア
170 xortemp &= ( ~( mask ) );
171 }
172 }
173 }
174 }
